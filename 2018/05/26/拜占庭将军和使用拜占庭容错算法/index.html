拜占庭将军问题PBFT实用拜占庭容错

2018-05-26 | 阅读：次

拜占庭将军问题

BFT 拜占庭将军问题 30个节点当中必须有21个是可用的,才能完成这项任务

客户端把信息”哈喽”给主节点A

A发布信息”哈喽”给B,C,D

B收到信息”哈喽”给A,C,D

C收到信息”哈喽”给A,B,D

D收到信息”哈喽”给A,B,C

PBFT 实用拜占庭容错 信息广播后ABCD每个节点都有收到的”哈喽”,再把”哈喽”返回给客户端, 客户端收到ABCD回复的信息如果有有2条以上的 信息为”哈喽”, 那最终结果就是”哈喽”

go来实现

package main

import (
	"os"
	"fmt"
	"net/http"
	"io"
)

//声明nodeInfo节点,代表各个小的王国
type nodeInfo struct {
	//节点名称
	id string
	//节点路径
	path string
	//http响应
	write http.ResponseWriter //跟终端进行交互
}

//创建map,存储各个国家的ip地址
var nodeTable = make(map[string]string)

func main() {
	//接收终端参数  先编译 然后运行./main 后面跟参数
	userID := os.Args[1] //1代表有一个参数
	fmt.Println(userID)
	//存储4个国家的Ip地址
	nodeTable = map[string]string{
		"Apple":  "localhost:11112",
		"MS":     "localhost:11113",
		"Google": "localhost:11114",
		"IBM":    "localhost:11115",
	}
	//创建国家对象
	node := nodeInfo{id: userID, path: nodeTable[userID]}
	//http协议的回调函数
	//http://localhost:1111/req?warTime=1111[0]&arTime=1112[1]
	http.HandleFunc("/req", node.request)
	//回调函数 通过/prePrepare
	http.HandleFunc("/prePrepare", node.prePrepare)
	//回调  通过/prepare
	http.HandleFunc("/prepare", node.prepare)
	http.HandleFunc("/commit",node.commit)

	//启动服务器  监听端口  如果有请求,会搜索handdleFunc里面的关键字
	 err := http.ListenAndServe(node.path, nil)
	if err != nil {
		fmt.Println(err)
	}

}
//writer http.ResponseWriter, request *http.Request固定格式
//http返回值					//https输入的值
//当http服务器,接收到网络请求并且/req 则回调request方法
func (node *nodeInfo) request(writer http.ResponseWriter, request *http.Request) {
	//该命令允许request请求参数
	//ParseForm解析URL中的查询字符串，并将解析结果更新到r.Form字段

	request.ParseForm() //取参数  ?后面的参数
	if (len(request.Form["warTime"]) > 0) {
		node.write = writer
		fmt.Println("主节点接收到的参数信息为", request.Form["warTime"][0])
		//激活主节点后,向其他节点发送广播
		node.broadcast(request.Form["warTime"][0], "/prePrepare") //广播给其他节点
	}
}

//节点发送广播的方法
func (node *nodeInfo) broadcast(msg string, path string) { //msg要发送的内容 path要发送的ip地址和端口
	fmt.Println("广播", path) //1133
	//遍历所有节点
	for nodeId, url := range nodeTable {
		if nodeId == node.id { //如果是本身, 就跳过这次
			continue
		}
		//使当前节点外的节点做响应
		//向指定的url发出请求
		http.Get("http://" + url + path + "?warTime=" + msg + "&nodeId=" + node.id)
	}
}

//接收广播后接收到的数据
func (node *nodeInfo) prePrepare(writer http.ResponseWriter, request *http.Request) {
	request.ParseForm() //啥意思?
	fmt.Println("接收到的广播", request.Form["warTime"][0])
	//做判断
	if len(request.Form["warTime"])>0{
		node.broadcast(request.Form["warTime"][0],"/prepare")

	}
}

//接收子节点的广播
func (node *nodeInfo) prepare(writer http.ResponseWriter, request *http.Request) {
	request.ParseForm()
	//打印消息
	fmt.Println("接收到的广播", request.Form["warTime"][0])
	//校验
	if len(request.Form["warTime"])>0{
		node.authentication(request)
	}

}

var authenticationNodeMap =make(map[string]string)
var authenticationSuceess = false

//校验拜占庭
func (node *nodeInfo)authentication(request *http.Request)  {
	if !authenticationSuceess{
		if len(request.Form["nodeId"])>0{
			authenticationNodeMap[request.Form["nodeId"][0]]="OK"
			//如果存了两个国家节点成功正确的返回了结果 就是true
			if len(authenticationNodeMap)>len(nodeTable)/3{
				authenticationSuceess=true
				node.broadcast(request.Form["warTime"][0],"/commit")
			}

		}
	}
}

//返回成功响应
func (node *nodeInfo)commit(writer http.ResponseWriter,request *http.Request)  {
	if writer!=nil{
		fmt.Println("拜占庭校验成功")
	}
	io.WriteString(node.write,"ok") //在网页显示
}
