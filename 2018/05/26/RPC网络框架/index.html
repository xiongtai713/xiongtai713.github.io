RPC网络框架

2018-05-26 | 阅读：次

RPC框架 用我电脑调用别的电脑里的函数 server 本台电脑 Area方法在Rect类型里 用其它的cilent来链接server,调用Rect里的Area方法

raft注册到rpc

rpc.Register(Rect)

rpc通过handle(路由器)可以通过网线或者无线把rpc到handle里面

rpc.HandleHTTP()

rpc监听本机
通过tcp协议把自己作为主服务器去监听别的电脑或者客户端,参数有port(本机ip地址和接口)和handle(回调函数,回调go语言库defaultfunc,去比较本机的ip地址和其他电脑拨打的ip地址是否一致)

http.ListenAndServe("IP地址:端口号","nil")

其他电脑进行拨打指定的ip地址及端口,如果和上边监听中的电脑ip地址一致,那么表示已经连上该机.

rpc1,err:=rpc.DialHTTP("TCP","要链接的IP地址:端口号")

go实现服务器

package main

import (
	"net/rpc"
	"net/http"
	"log"
)

//定义参数类型
type Params struct {
	Width, Height int
}

//声明对象
type Rect struct {
}

//定义公共方法 该方法在本机以外的方法也能调用
func (r *Rect) Area(p Params, ret *int) error {  //别人要调我的计算机的内存 所以要用指针
	*ret=p.Height*p.Width
	return nil
}

func (r *Rect)Perimtter(p Params, ret *int) error {
	*ret=(p.Width+p.Height)*2
	return nil

}

func main()  {
	//注册服务
	rect:=new(Rect)
	rpc.Register(rect)
	rpc.HandleHTTP() //不知道
	err:=http.ListenAndServe(":9090",nil)  //监听端头
	if err!=nil{
		log.Fatal(err)  //打印错误 并且写入日志
	}

}

链接成功后返回rpc1,可以调用Call方法来调用server中的communication方法

rpc1.Call("要调用的方法路径Rect.Area","要传入方法的参数","方法给的结果&")

go实现客户端

package main

import (
	"net/rpc"
	"log"
	"fmt"
)

//实现rpc客户端

type Params struct { //客户端要跟服务器保持一致
	Width, Height int
}

//调用服务器
func main() {
	//链接远程rpc
	rpc1,err:=rpc.DialHTTP("tcp","127.0.0.1:9090")//链接
	if err!=nil{
		log.Fatal(err)
	}
	ret:=0  //接收服务器传递的结果
	e:=rpc1.Call("Rect.Area",Params{100,100},&ret)

	if e!=nil{
		log.Fatal(e)
	}
	fmt.Println(ret)
}

结果就是ret=100
