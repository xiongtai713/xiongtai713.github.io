json数据处理

3.1 编码JSON

使用json.Marshal()函数可以对一组数据进行JSON格式的编码。

 json.Marshal()函数的声明如下：

func Marshal(v interface{}) ([]byte, error)

还有一个格式化输出：

// MarshalIndent 很像 Marshal，只是用缩进对输出进行格式化

func MarshalIndent(v interface{}, prefix, indent string) ([]byte, error)

编码JSON

3.1.1 通过结构体生成json

实例代码

package main

import (
	"encoding/json"
	"fmt"
)

type person struct {
	First string  //正常
	Last  string `json:"-"`  //加了json标签"-"代表忽略
	Age   int    `json:"wisdom score"`// 填加json标签 在json字符串里显示
}

func main() {
	p1 := person{"James", "Bond", 20}
	bs, _ := json.Marshal(p1)
	fmt.Println(string(bs))
}

注意：结构体的字段(首字母是大写)才会被输出

关于 _ 、别名、类型转换、空数据的使用

1、_ 可做忽略对应的字段

2、 json:"subjects" json：‘’ 取别名完成映射

3、json:",string" 类型转换

4、 json:"price, omitempty"

3.1.2 通过map生成JSON

package main

import (
	"encoding/json"
	"fmt"
)

func main () {
	//创建一个map
	m := make(map[string]interface{}, 4)
	m["company"] = "我爱编程"
	m["subjects"] = []string{"语文", "数学", "英语", "化学"}
	m["isok"] = true
	m["price"] = 1.1

	//编码成json
	result, err := json.MarshalIndent(m, "", "	")
	if err != nil {
		fmt.Println("err = ", err)
		return
	}
	fmt.Println("result = ", string(result))
}

3.2 解码JSON

可以使用json.Unmarshal()函数将JSON格式的文本解码为Go里面预期的数据结构。

3.2.1 json字符串解析到结构体

package main

import (
	"encoding/json"
	"fmt"
)

type Test02 struct {
	Company  string   `json:"company"`
	Subjects []string `json:"subjects"` //二次编码
	IsOk     bool     `json:"isok"`
	Price    float64  `json:"price"`
}

func main() {

	jsonBuf := `
	{
    "company": "我爱编程",
    "subjects": [
        "语文",
        "数学",
        "英语",
        "化学"
    ],
    "isok": true,
    "price": 1.1
}`
	//定义一个结构体变量
	var tmp Test02
	err := json.Unmarshal([]byte(jsonBuf), &tmp) //第二个参数要地址传递
	if err != nil {
		fmt.Println("err = ", err)
		return
	}
	//fmt.Println("tmp = ", tmp)
	fmt.Printf("tmp = %+v\n", tmp)

	type Test03 struct {
		Subjects []string `json:"subjects"` //二次编码
	}

	var tmp2 Test03
	err = json.Unmarshal([]byte(jsonBuf), &tmp2) //第二个参数要地址传递
	if err != nil {
		fmt.Println("err = ", err)
		return
	}
	fmt.Printf("tmp2 = %+v\n", tmp2)

}

3.2.1 json字符串解析到Map集合中

package main

import (
	"encoding/json"
	"fmt"
)

func main() {
    //定义json字符串
	jsonBuf := `
	{
    "company": "我爱编程",
    "subjects": [
        "语文",
        "数学",
        "英语",
        "化学"
    ],
    "isok": true,
    "price": 1.1
	}`

	//创建一个map
	m := make(map[string]interface{}, 4)

	err := json.Unmarshal([]byte(jsonBuf), &m) //第二个参数要地址传递
	if err != nil {
		fmt.Println("err = ", err)
		return
	}
	fmt.Printf("m = %+v\n", m)

	//	var str string
	//	str = string(m["company"]) //err， 无法转换
	//	fmt.Println("str = ", str)


	//遍历map集合，判断类型
	var str string

	//类型断言, 值，它是value类型
	for key, value := range m {
		//fmt.Printf("%v ============> %v\n", key, value)
		switch data := value.(type) {
		case string:
			str = data
			fmt.Printf("map[%s]的值类型为string, value = %s\n", key, str)
		case bool:
			fmt.Printf("map[%s]的值类型为bool, value = %v\n", key, data)
		case float64:
			fmt.Printf("map[%s]的值类型为float64, value = %f\n", key, data)
		case []string:
			fmt.Printf("map[%s]的值类型为[]string, value = %v\n", key, data)
		case []interface{}:
			fmt.Printf("map[%s]的值类型为[]interface, value = %v\n", key, data)
		}

	}

}

3.3 Decoder和Encoder 的使用

Encode：对象转换成json字符串

package main

import (
	"encoding/json"
	"os"
)

type person struct {
	Name string
	Age string
	sex string
}

func main () {
	p1 := person{"liubei","18","nan"}

	json.NewEncoder(os.Stdout).Encode(p1)
}

//{"Name":"liubei","Age":"18"}

Decoder: json字符串转对象

package main

import (
	"encoding/json"
	"strings"
	"fmt"
)

type person01 struct {
	Name string
	Age string
	sex string
}

func main () {
	var p2 person01

	str := strings.NewReader(`{"Name":"lixunhuan","Age":"22"}`)

	json.NewDecoder(str).Decode(&p2)

	fmt.Println(p2)
}
//{lixunhuan 22 }
