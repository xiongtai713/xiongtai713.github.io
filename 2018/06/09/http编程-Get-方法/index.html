go网络编程GET

2018-06-09 | 阅读：次

http编程 Get 方法

客户端给服务器发送请求,服务器根据请求给予回应 go有三种Get请求方法

###第一种直接GET 这种方法直接接收服务器的回应,得到服务器回应后,把body里的数据读取出来

package main

import (
	"net/http"
	"sanqi/0604socket/help"
	"fmt"
	"io/ioutil"
)

func main()  {
	//获取服务器的数据
	resp, err := http.Get("https://www.baidu.com")
	help.CheckErr(err)
	//fmt.Printf("%+v\n",resp)
	if resp.StatusCode==200{  //判断是否是200
		//操作响应数据
		defer resp.Body.Close() //关闭
		//读取body里面的数据
		bytes, err := ioutil.ReadAll(resp.Body)  //读取body中的reader方法
		help.CheckErr(err)

		//打印响应体里面的数据
		fmt.Println(string(bytes))  //body是个接口

	}else {
		fmt.Println("请求失败",resp.Status) //打印状态码
	}
}

###第二种新建请求 先新建客户端(client),在新建请求(NewRequest) 在通过client.Do(Request)请求服务器给予回复 好处就是可以先编辑请求,然后再发出

package main

import (
	"net/http"
	"sanqi/0604socket/help"
	"fmt"
	"io/ioutil"
)

func main()  {
	//1.创建一个客户端
	client:=http.Client{}
	//2.创建一个请求
	request, err := http.NewRequest("GET", "https://www.toutiao.com/search/suggest/initial_page/", nil) //新建一个请求
	if err!=nil{
		fmt.Println("get数据失败",err)
	}
	//添加cookie
	fmt.Println(request)
	cookieName := http.Cookie{Name: "liujiao", Value: "hahaha"}
	request.AddCookie(&cookieName) //把创建的cookie放到请求里取去
	//3.客户端发送请求
	response, err := client.Do(request)  //把请求发送到url网址 得到服务器响应
	//设置请求头
	request.Header.Set("Accept-Lanauage","zh-cn")

	//查看请求头
	fmt.Printf("Header%+v\n",request.Header)
	//requestBody,err:=ioutil.ReadAll(request.Body)
	//fmt.Printf("requestBody%+v\n",requestBody)


	if err!=nil{
		fmt.Println("客户端发送请求失败",err)
	}
	defer response.Body.Close()  //关闭
	fmt.Printf("%v\n%v\n",response.StatusCode,response.Header)
	//4.操作数据
	if  response.StatusCode==http.StatusOK{
		bytes, err := ioutil.ReadAll(response.Body)
		help.CheckErr(err)
		fmt.Println(string(bytes))


	}else {
		fmt.Println("网络请求失败",response.Status)
	}
}

###第三种新建客户端直接GET 和直接http.get差不多

package main

import (
	"net/http"
	"sanqi/0604socket/help"
	"io/ioutil"
	"fmt"
)

func main()  {
	//创建客户端
	client := http.Client{}
	//通过client去请求
	response, err := client.Get("https://www.toutiao.com/search/suggest/initial_page/")
	help.CheckErr(err)
	if response.StatusCode==http.StatusOK{
		defer response.Body.Close()
		bytes, err := ioutil.ReadAll(response.Body)
		if err!=nil{
			fmt.Println(err)
		}
		fmt.Println(string(bytes))   //打印请求body
		fmt.Println(response.Header)  //打印请求头
	}

}

GET还可以通过路径来请求服务器端的方法 http://localhost:9527/hi?theCityName=北京 正常url后面跟的是调用方法名称/hi theCityName是map的k,通过服务器通过r.ParseForm()可以得到value =后面是map的value

package main

import (
	"net/http"
	"fmt"
	"io/ioutil"
)

func main()  {
	resp, _ := http.Get("http://localhost:9527/hi?theCityName=北京")
	bytes, _ := ioutil.ReadAll(resp.Body)
	fmt.Println(string(bytes))

}

###服务器端 http.HandleFunc通过客户端发来的调用方法的名称,来调用对应的方法, func ServeHTTP3(w http.ResponseWriter, r *http.Request)两个参数是固定写法,w代表给客户端回复的数据,r代表客户端传来的数据, 可以通过r.ParseForm()来解析客户端发来的数据 r.form 是给map,可以通过客户端传来的k来取里面的v,通过v来做判断给客户端回复相对应的数据 w.Write可以直接写数据给客户端

package main

import (
	"net/http"
	"fmt"
)
func ServeHTTP3(w http.ResponseWriter, r *http.Request) {
	r.ParseForm()//解析客户端发送过来的参数
	fmt.Println(r.FormValue("theCityName"))
	if len(r.FormValue("theCityName"))>0{
		fmt.Println(r.FormValue("theCityName")[0])  //通过k来找value
		if r.FormValue("theCityName")=="北京"{
			responsrstr:=`<ArrayOfString xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance" xmlns:xsd="http://www.w3.org/2001/XMLSchema" xmlns="http://WebXml.com.cn/">
  <string>访问被限制！</string>
	w.Write([]byte(responsrstr))
		}
	}
	fmt.Println(r.Header) //打印请求头
	fmt.Println(r.Method) //打印请求方法
	fmt.Println(r.Cookie) //打印客户端cookie
	fmt.Println(r.URL)
	if r.Method=="GET"{
		w.Write([]byte("不安全"))
	}
	w.Write([]byte("第一个go后台程序"))
}
func ServeHTTP4(w http.ResponseWriter, r *http.Request) {
	w.Write([]byte("哈哈"))
}
func ServeHTTP5(w http.ResponseWriter, r *http.Request) {
	w.Write([]byte("喵喵"))
}

func main()  {
	mux := http.NewServeMux()  //新建mux
	mux.HandleFunc("/hi",ServeHTTP3)  //用mux来监听绑定
	mux.HandleFunc("/ha",ServeHTTP5)
	mux.HandleFunc("/miao",ServeHTTP4)

	http.ListenAndServe(":9527",mux)
	
}
