go网络编程总结

2018-06-09 | 阅读：次

总结

###Socket 服务端 (tcp或udp)

net.Listen(“tcp”,”ip:port”) 监听 –> conn:= lister.Accept() 等待链 接–>conn read||conn.Write 或者写

###Socket 客户端 (tcp或udp)

客户端 conn:=net.Dial(“tcp”,”ip:port”) 链接服务器 –> conn read   conn.Write 读或写

###rpc传输 服务器(1)调用服务器(2)的方法 ###服务器(1) 调用方

client, err := rpc.DialHTTP(“tcp”, “:8888”) //链接服务器(2) client.Call(“MyMath.Add”, Param{23,9}, &sum) //调用服务器(2)的MyMath.Add方法 Param是参数1(方法调用所需的参数), sum是参数2(服务器2给返回的结果,所以一定要传指针)

###服务器(2) 被调用方

//注册的结构体,将方法绑定到RPC服务
type MyMath struct {
}
//提供方法所需要的参数
type Param struct {  //因为只能接收一个值 所有要创建结构体
	A int
	B int
}
//这是提供给客户端调用的方法
func (m *MyMath) Add(param Param,sum *int)error  {
	*sum=param.A+param.B
	return nil
}

rpc.Register(new(MyMath)) //把结构体注册到rpc 等待调用 rpc.HandleHTTP() //把rpc绑定到http里 http.ListenAndServe(“:8888”,nil) //通过http来监听

###http协议 客户端 GET 获取网页内容

resp, err := http.Get(“https://www.baidu.com”) //获取网页内容, bytes, err := ioutil.ReadAll(resp.Body) //网页的内容全在body里,可以读取出来 ,bytes就是读取出的内容

###http协议 客户端 GET 请求服务端

resp, _ := http.Get(“http://localhost:9527/hi?theCityName=北京”) //?后面是请求服务器的参数,服务器会根据参数给予不同的回应 bytes, _ := ioutil.ReadAll(resp.Body) //服务器给予的回应都在body里,可以读取出来

###http协议 客户端 POST 请求服务端

param:=url.Values{  //post请求是通过body来给客户端传递参数的,所以要创建body "theCityName":{"北京"}, }   //buffer实现了io.Reader,所以创建一个buffer,可以当body用 buffer :=bytes.NewBufferString(param.Encode())  //转换成string //POST请求有三个参数,1.url/路径 2.编码格式, 3 请求body response, _ := http.Post("http://localhost:9527/hi","application/x-www-form-urlencoded", buffer)

###http协议 服务端

>http.HandleFunc("/hi",ServeHTTP)//pattern是url的路径 域名后面的路径
>//如果客户端发来hi,就调用ServeHTTP方法给予客户端回复
>http.ListenAndServe(":9527",nil)  //监听9527端口
>//下面是ServeHTTP方法
>func ServeHTTP(w http.ResponseWriter, r *http.Request) {
	r.ParseForm()//解析客户端发送过来的参数
	fmt.Println(r.Form["theCityName"][0])
	fmt.Println(r.FormValue("theCityName"))
	if len(r.FormValue("theCityName"))>0{
		fmt.Println(r.FormValue("theCityName")[0])  //通过k来找value
		if r.FormValue("theCityName")=="北京"{
			responsrstr:=`<ArrayOfString xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance" xmlns:xsd="http://www.w3.org/2001/XMLSchema" xmlns="http://WebXml.com.cn/">

`
	w.Write([]byte(responsrstr))
		}
	}
	fmt.Println(r.Header) //打印请求头
